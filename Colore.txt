
import java.util.Random;
import javafx.application.Application;
import javafx.event.ActionEvent;
import javafx.event.EventHandler;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class ParoleColori extends Application implements EventHandler<ActionEvent> {

    public static String[] colore_parole = {"Nero", "Rosso", "Verde", "Giallo", "Blu", "Viola", "Azzurro", "Bianco"};
    public static String[] colore_ansi = {"\033[0;30m", "\033[0;31m", "\033[0;32m", "\033[0;33m", "\033[0;34m", "\033[0;35m", "\033[0;36m", "\033[0;37m"};
    Button genera;
    Button pulisci;
    public static void Stampa() {
        int num1 = 0;
        int num2 = 0;
        Random rand = new Random();

        for (int i = 0; i < 10; i++) {
            num1 = rand.nextInt(8);
            num2 = rand.nextInt(8);
            if (num1 == num2) {
                while (num1 != num2) {
                    num1 = rand.nextInt(8);
                    num2 = rand.nextInt(8);
                }
            }
            System.out.print(colore_parole[num1] + colore_ansi[num2] + " ");
        }
    }//fine Stampa

    public static void clearScreen() {
        for (int i = 0; i < 50; ++i) 
            System.out.println();
    }//fine clearScreen

    public static void main(String[] args) {
        launch(args);
    }//fine main

    @Override
    public void start(Stage primarystage) throws Exception {
        primarystage.setTitle("Parole & Colori");
        
        genera = new Button("Genera parole di colori casuali");
        genera.setOnAction(this);
        pulisci = new Button("Pulisci");
        pulisci.setOnAction(this);
        
        StackPane layout = new StackPane();
        layout.getChildren().add(genera);
        layout.getChildren().add(pulisci);
        
        Scene scene = new Scene(layout, 300, 300);
        primarystage.setScene(scene);

        primarystage.show();
    }//fine start

    public void handle(ActionEvent event) {
        if (event.getSource() == genera) {
            Stampa();            
        }
        if(event.getSource() == pulisci){
            clearScreen();
        }
    }

}//fine class
